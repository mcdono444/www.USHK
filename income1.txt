function getFirebase() {
    return new Promise((resolve) => {
        const check = setInterval(() => {
            if (window.firebaseModules) {
                clearInterval(check);
                resolve(window.firebaseModules);
            }
        }, 100);
    });
}

function showMessage(msg, type) {
    var div = $('<div class="toast ' + type + '">' + msg + '</div>');
    $('body').append(div);
    setTimeout(function() { div.fadeOut(300, function() { $(this).remove(); }); }, 3000);
}

// ✅ OBTER CONVIDADOS DA COLEÇÃO /referrals/{userId}/invited
async function getMyInvited(db, userId) {
    try {
        console.log('🔍 Buscando en /referrals/' + userId + '/invited');
        const invitedRef = collection(db, 'referrals', userId, 'invited');
        const snaps = await getDocs(invitedRef);
        
        const invited = snaps.docs.map(d => ({ 
            id: d.id, 
            ...d.data() 
        }));
        
        console.log('✅ Documentos encontrados:', invited.length);
        invited.forEach((inv, idx) => {
            console.log(`  [${idx}]`, inv);
        });
        
        return invited;
    } catch (err) {
        console.error('❌ Erro ao buscar convidados:', err);
        return [];
    }
}

// ✅ ENRIQUECER DADOS DO CONVIDADO
async function enrichInvited(db, invited) {
    const out = Object.assign({}, invited);
    
    try {
        const invitedUserId = invited.invitedUserId || invited.id;
        
        if (invitedUserId) {
            const userDocRef = doc(db, 'users', invitedUserId);
            const userSnap = await getDoc(userDocRef);
            
            if (userSnap.exists()) {
                const d = userSnap.data();
                out.phone = d.phone || invited.invitedPhone || d.email || invited.invitedEmail || null;
                out.email = d.email || invited.invitedEmail || null;
                out.createdAt = d.createdAt || invited.invitedAt || null;
                out.vipLevel = d.vipLevel || 0;
                out.hasPurchased = d.hasPurchased || false;
                out.balance = d.balance || 0;
                console.log('✅ Datos enriquecidos para:', invitedUserId);
            } else {
                out.phone = invited.invitedPhone || invited.invitedEmail || null;
                out.email = invited.invitedEmail || null;
                out.createdAt = invited.invitedAt || null;
                console.log('⚠️ Usuario no encontrado:', invitedUserId);
            }
        }
    } catch (e) {
        console.warn('⚠️ Error al enriquecer:', e);
        out.phone = invited.invitedPhone || invited.invitedEmail || null;
        out.email = invited.invitedEmail || null;
        out.createdAt = invited.invitedAt || null;
    }
    
    return out;
}

// ✅ CARREGAR DATOS DEL EQUIPO
async function loadTeamData() {
    try {
        const { db, doc: firebaseDoc, getDoc, collection, getDocs } = await getFirebase();

        console.log('\n📊 === INICIANDO CARGA DE EQUIPO ===');
        console.log('User ID:', userId);

        // OBTER DATOS DEL USUÁRIO ATUAL
        const userRef = firebaseDoc(db, 'users', userId);
        const userSnap = await getDoc(userRef);

        if (!userSnap.exists()) {
            showMessage('Error: Usuario no encontrado', 'error');
            console.error('❌ Usuario no encontrado');
            return;
        }

        const userData = userSnap.data();
        const inviteCode = userData.referralCode || userData.phone || userId.substring(0, 8);
        const inviteLink = window.location.origin + '/register.html?ref=' + encodeURIComponent(inviteCode);

        console.log('✅ Usuario encontrado');
        console.log('  Código:', inviteCode);

        $('#inviteCode').text(inviteCode);
        $('#inviteLink').text(inviteLink);

        // ✅ NIVEL 1: CONVIDADOS DIRETOS
        console.log('\n📍 Cargando Nivel 1...');
        const rawLevel1 = await getMyInvited(db, userId);
        const level1Members = [];
        let level1ActiveCount = 0;

        for (const r of rawLevel1) {
            const member = await enrichInvited(db, r);
            member.userId = member.invitedUserId || member.id;
            member.joinAt = member.createdAt || member.invitedAt || null;
            
            const isActive = member.hasPurchased === true || member.hasPurchased === 'true' || 
                           (member.vipLevel && Number(member.vipLevel) > 0);
            if (isActive) level1ActiveCount++;
            
            level1Members.push(member);
        }

        console.log('✅ Nivel 1 completado:', level1Members.length, 'miembros, activos:', level1ActiveCount);

        // ✅ NIVEL 2
        console.log('\n📍 Cargando Nivel 2...');
        const level2Members = [];
        let level2ActiveCount = 0;

        for (const m1 of level1Members) {
            if (!m1.userId) continue;
            
            const raw2 = await getMyInvited(db, m1.userId);
            for (const r2 of raw2) {
                const member2 = await enrichInvited(db, r2);
                member2.userId = member2.invitedUserId || member2.id;
                member2.parentId = m1.userId;
                member2.joinAt = member2.createdAt || member2.invitedAt || null;
                
                const isActive = member2.hasPurchased === true || member2.hasPurchased === 'true' || 
                               (member2.vipLevel && Number(member2.vipLevel) > 0);
                if (isActive) level2ActiveCount++;
                
                level2Members.push(member2);
            }
        }

        console.log('✅ Nivel 2 completado:', level2Members.length, 'miembros, activos:', level2ActiveCount);

        // ✅ NIVEL 3
        console.log('\n📍 Cargando Nivel 3...');
        const level3Members = [];
        let level3ActiveCount = 0;

        for (const m2 of level2Members) {
            if (!m2.userId) continue;
            
            const raw3 = await getMyInvited(db, m2.userId);
            for (const r3 of raw3) {
                const member3 = await enrichInvited(db, r3);
                member3.userId = member3.invitedUserId || member3.id;
                member3.parentId = m2.userId;
                member3.joinAt = member3.createdAt || member3.invitedAt || null;
                
                const isActive = member3.hasPurchased === true || member3.hasPurchased === 'true' || 
                               (member3.vipLevel && Number(member3.vipLevel) > 0);
                if (isActive) level3ActiveCount++;
                
                level3Members.push(member3);
            }
        }

        console.log('✅ Nivel 3 completado:', level3Members.length, 'miembros, activos:', level3ActiveCount);

        teamData = {
            level1: level1Members,
            level1Active: level1ActiveCount,
            level2: level2Members,
            level2Active: level2ActiveCount,
            level3: level3Members,
            level3Active: level3ActiveCount
        };

        // ATUALIZAR TOTAIS
        const totalMembers = level1Members.length + level2Members.length + level3Members.length;
        const activeInvestors = level1ActiveCount + level2ActiveCount + level3ActiveCount;

        console.log('\n📊 TOTALES:');
        console.log('  Total miembros:', totalMembers);
        console.log('  Activos:', activeInvestors);

        $('#totalInvited').text(totalMembers);
        $('#activeInvestors').text(activeInvestors);

        // ATUALIZAR CONTADORES NAS ABAS
        $('#tab1Count').text('(' + level1Members.length + ')');
        $('#tab2Count').text('(' + level2Members.length + ')');
        $('#tab3Count').text('(' + level3Members.length + ')');

        // ✅ CALCULAR COMISSÕES
        console.log('\n💰 Buscando comisiones...');
        let totalCommission = 0;
        try {
            const balancesRef = collection(db, 'balances');
            const q = query(balancesRef, where('userId', '==', userId), where('type', '==', 'referral_bonus'));
            const snaps = await getDocs(q);
            
            console.log('  Registros encontrados:', snaps.size);
            snaps.forEach(s => {
                const t = s.data();
                totalCommission += Number(t.amount || 0);
            });
            
            console.log('✅ Comisión total:', totalCommission);
        } catch (e) {
            console.warn('⚠️ Error al buscar comisiones:', e);
        }

        $('#totalCommission').text(totalCommission.toFixed(0));
        $('#totalCommissionAmount').text(totalCommission.toFixed(2) + ' kz');

        console.log('\n✅ === CARGA COMPLETA ===\n');
        renderTeamList();
        
    } catch (error) {
        console.error('❌ ERROR GENERAL:', error);
        showMessage('Error: No se pudieron cargar los datos', 'error');
        $('#teamList').html(`
            <div class="empty-team">
                <i class="fas fa-wifi"></i>
                <p>Error de conexión</p>
                <p style="font-size: 12px;">Por favor, intente novamente</p>
                <div class="debug-info">Error: ${error.message}</div>
            </div>
        `);
    }
}

function renderTeamList() {

    let levelName = '';

    if (currentLevel === 1) {
        members = teamData.level1 || [];
        levelName = 'Nivel 1';
    } else if (currentLevel === 2) {
        members = teamData.level2 || [];
        levelName = 'Nivel 2';
    } else {
        members = teamData.level3 || [];
        levelName = 'Nivel 3';
    }

    if (members.length === 0) {
        $('#teamList').html(`
            <div class="empty-team">
                <i class="fas fa-users"></i>
                <p>No hay miembros en ${levelName}</p>
                <p style="font-size: 12px;">Invita a tus amigos para construir tu equipo</p>
            </div>
        `);
        return;
    }

    let html = '';
    
    members.forEach((member, index) => {
        const isActive = member.hasPurchased === true || member.hasPurchased === 'true' || 
                        (member.vipLevel && Number(member.vipLevel) > 0);
        const statusClass = isActive && member.vipLevel && Number(member.vipLevel) > 0 ? 'badge-vip' : 
                           isActive ? 'badge-active' : 'badge-inactive';
        const statusText = isActive && member.vipLevel && Number(member.vipLevel) > 0 ? `VIP${member.vipLevel} ✓` : 
                          isActive ? '✓ Activo' : '✗ Inactivo';

        let joinDate = 'Fecha no disponible';
        if (member.joinAt) {
            try {
                const d = new Date(member.joinAt);
                if (!isNaN(d.getTime())) {
                    joinDate = d.toLocaleDateString('es-MZ') + ' ' + d.toLocaleTimeString('es-MZ', { hour: '2-digit', minute: '2-digit' });
                }
            } catch (e) {
                joinDate = String(member.joinAt).substring(0, 10);
            }
        }

        const displayPhone = member.phone || member.email || member.invitedPhone || member.invitedEmail || 'Sin contacto';
        const displayEmail = member.email || member.invitedEmail || '';

        html += `
            <div class="team-member" style="animation-delay: ${index * 0.05}s;">
                <div class="member-info">
                    <div class="member-phone">📱 ${displayPhone}</div>
                    <div class="member-date">
                        <i class="fas fa-calendar-alt"></i> ${joinDate}
                        ${displayEmail ? `<br><i class="fas fa-envelope"></i> ${displayEmail}` : ''}
                    </div>
                </div>
                <div class="member-status">
                    <div class="member-badge ${statusClass}">${statusText}</div>
                </div>
            </div>
        `;
    });
    
    $('#teamList').html(html);
}

function copyInviteCode() {
    const code = $('#inviteCode').text();
    if (code && code !== 'Cargando...') {
        navigator.clipboard.writeText(code).then(function() {
            showMessage('✅ ¡Código copiado!', 'success');
        }).catch(function() {
            const textarea = document.createElement('textarea');
            textarea.value = code;
            document.body.appendChild(textarea);
            textarea.select();
            document.execCommand('copy');
            document.body.removeChild(textarea);
            showMessage('✅ ¡Código copiado!', 'success');
        });
    }
}
